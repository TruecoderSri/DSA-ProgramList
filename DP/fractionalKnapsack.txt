 class Solution
{

 // Greedy approach to get the max profit knapsack by greddily sorting based on profit/wt ratio and then extracting ele one-by-one

    public:
    static bool compare(Item a,Item b){
        double f1=(double) a.value/a.weight;
        double f2=(double) b.value/b.weight;
        return f1>f2;
    }
    
    double fractionalKnapsack(int w, Item arr[], int n)
    {
        sort(arr,arr+n,compare);
        double ans=0;
        for(int i=0;i<n;i++){
            int val=arr[i].value;
            int wei=arr[i].weight;
            if(wei<w){
                ans+=(double)val;
                w-=wei;
            }
            else{
                ans+=(double)  val/wei *w;        // get fraction of the weight and then bag is full because wt was already >
                w=0;
                break;
            }
        }
        
        
        return ans;
        
    }
